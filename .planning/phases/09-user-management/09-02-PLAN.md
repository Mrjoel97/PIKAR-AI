From app/agents/admin/tools/monitoring.py (_check_autonomy pattern to copy):
```python
async def _check_autonomy(action_name: str) -> dict | None:
    """Query admin_agent_permissions and return a gate dict if blocked/confirm."""
    try:
        client = get_service_client()
        res = (
            client.table("admin_agent_permissions")
            .select("autonomy_level")
            .eq("action_name", action_name)
            .limit(1)
            .execute()
        )
        if res.data:
            level = res.data[0].get("autonomy_level", "auto")
            if level == "blocked":
                return {"error": f"{action_name} is blocked ..."}
            if level == "confirm":
                token = str(uuid.uuid4())
                return {
                    "requires_confirmation": True,
                    "confirmation_token": token,
                    "action_details": {
                        "action": action_name,
                        "risk_level": "low",
                        "description": "...",
                    },
                }
    except Exception as exc:
        logger.warning("Could not verify autonomy level ...: %s", exc)
    return None
```
