    **frontend/src/lib/sanitize.ts** -- Create a centralized DOMPurify wrapper:
    ```typescript
    import DOMPurify from 'dompurify';

    /**
     * Sanitize HTML content using DOMPurify.
     * Strips all dangerous elements (scripts, event handlers, javascript: URIs, SVG-based XSS).
     */
    export function sanitizeHtml(dirty: string): string {
      return DOMPurify.sanitize(dirty, {
        USE_PROFILES: { html: true },
        FORBID_TAGS: ['style'],
        FORBID_ATTR: ['style'],
      });
    }
    ```

    **frontend/src/app/p/[id]/page.tsx** -- Remove the hand-rolled sanitizeHtml function (lines 7-21) and the comment about trusting agent-generated HTML (line 56). Replace with import from the new shared utility:

    The existing call to sanitizeHtml on line 77 stays the same — it now uses DOMPurify under the hood. This is the proper fix: DOMPurify handles data: URIs, SVG vectors, CSS injection, mutation XSS, and all known bypass techniques that the hand-rolled version misses.

    **app/routers/community.py** -- Add Pydantic Field constraints to request models:
    ```python
    from pydantic import BaseModel, Field, field_validator

    class CreatePostRequest(BaseModel):
        title: str = Field(..., min_length=1, max_length=200)
        body: str = Field(..., min_length=1, max_length=10_000)
        category: str | None = Field(default="general", max_length=50)
        tags: list[str] | None = Field(default=None, max_length=10)

        @field_validator("tags")
        @classmethod
        def validate_tag_lengths(cls, v: list[str] | None) -> list[str] | None:
            if v is None:
                return v
            for tag in v:
                if len(tag) > 50:
                    raise ValueError("Each tag must be 50 characters or fewer")
                if not tag.strip():
                    raise ValueError("Tags must not be empty")
            return [t.strip() for t in v]

    class CreateCommentRequest(BaseModel):
        body: str = Field(..., min_length=1, max_length=5_000)
    ```
    Import field_validator from pydantic alongside BaseModel and Field.
