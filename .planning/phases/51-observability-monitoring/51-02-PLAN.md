1. **Create a helper function** (place it near the health endpoints, around line 1040):

```python
def _health_response(
    *,
    status: str,
    service: str,
    latency_ms: int | None = None,
    details: dict | None = None,
    integrations: dict | None = None,
) -> dict:
    """Build canonical health response envelope (version 1)."""
    from datetime import datetime, timezone
    resp: dict = {
        "status": status,
        "version": "1",
        "service": service,
        "latency_ms": latency_ms,
        "details": details or {},
        "checked_at": datetime.now(timezone.utc).isoformat(),
    }
    if integrations is not None:
        resp["integrations"] = integrations
    return resp
```

2. **`/health/live`** — Wrap existing response:
   - `status="ok"`, `service="live"`, `latency_ms=0`, `details={}`.

3. **`/health/connections`** — This is the most complex one. Preserve all existing checks (service client connectivity, RAG client connectivity, env var checks). Wrap the final response:
   - On success: `status="ok"`, `service="supabase"`, compute `latency_ms` from the connectivity check timing, `details` includes the existing supabase stats, env info.
   - Add an `integrations` subkey: query `integration_sync_state` table using the service client, build a dict like `{"hubspot": {"status": "ok", "last_sync_at": "..."}, ...}`. Map `last_sync_status` to status (any status containing "error" or "failed" maps to "degraded", null maps to "unknown", else "ok"). This is zero extra API calls — reads cached state from the DB.
   - On failure: `status="down"`, include error in details.

4. **`/health/cache`** — Wrap existing Redis circuit breaker check:
   - `service="redis"`, preserve existing details (connected, circuit_breaker state).
   - `status="ok"` when connected and circuit closed, `status="degraded"` when circuit open but app running, `status="down"` on exception.

5. **`/health/embeddings`** — Wrap existing Gemini embedding check:
   - `service="gemini"`, preserve existing details.
   - `status="ok"` on success, `status="down"` on failure.

6. **`/health/video`** — Wrap existing video config check:
   - `service="video"`, preserve existing details (generation config).
   - `status="ok"` when properly configured, `status="degraded"` when partially configured.

IMPORTANT: The existing response bodies had inconsistent shapes. The new shape is uniform. Existing consumers (`health_checker.py` and `monitoring.py`) parse these — they will be updated in Task 2. Since deployment is atomic (monorepo), this is safe.

Keep the existing `/health/startup` and `/health/workflows/readiness` endpoints unchanged — they are not in the HEALTH_ENDPOINTS registry and are consumed by Cloud Run startup probes only.

"""Tests for canonical health endpoint response shape (OBS-05)."""

import pytest
from unittest.mock import MagicMock, patch, AsyncMock
import os


class TestCanonicalHealthShape:
    """Verify all /health/* endpoints return the canonical versioned envelope."""

    REQUIRED_KEYS = {"status", "version", "service", "latency_ms", "details", "checked_at"}
    VALID_STATUSES = {"ok", "degraded", "down"}

    def test_health_response_helper_produces_all_fields(self):
        """_health_response includes all required canonical fields."""
        # Import inside test to avoid module-level side effects
        with patch.dict(os.environ, {
            "SUPABASE_URL": "https://test.supabase.co",
            "SUPABASE_SERVICE_ROLE_KEY": "test-key",
            "SUPABASE_ANON_KEY": "test-anon",
        }):
            from app.fast_api_app import _health_response
            resp = _health_response(status="ok", service="test", latency_ms=42, details={"foo": "bar"})
            assert self.REQUIRED_KEYS.issubset(resp.keys())
            assert resp["version"] == "1"
            assert resp["status"] in self.VALID_STATUSES

    def test_health_response_status_constrained(self):
        """Status values must be ok, degraded, or down."""
        with patch.dict(os.environ, {
            "SUPABASE_URL": "https://test.supabase.co",
            "SUPABASE_SERVICE_ROLE_KEY": "test-key",
            "SUPABASE_ANON_KEY": "test-anon",
        }):
            from app.fast_api_app import _health_response
            for status in self.VALID_STATUSES:
                resp = _health_response(status=status, service="test")
                assert resp["status"] == status

    def test_health_response_integrations_optional(self):
        """integrations key only present when explicitly provided."""
        with patch.dict(os.environ, {
            "SUPABASE_URL": "https://test.supabase.co",
            "SUPABASE_SERVICE_ROLE_KEY": "test-key",
            "SUPABASE_ANON_KEY": "test-anon",
        }):
            from app.fast_api_app import _health_response
            resp_without = _health_response(status="ok", service="test")
            assert "integrations" not in resp_without
            resp_with = _health_response(status="ok", service="test", integrations={"stripe": {"status": "ok"}})
            assert "integrations" in resp_with


class TestHealthCheckerCanonicalParsing:
