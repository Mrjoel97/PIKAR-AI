```python
"""Centralized feature gating configuration.

Single source of truth for backend tier-to-feature access.
Must stay in sync with frontend/src/config/featureGating.ts.
"""

from __future__ import annotations

TIER_ORDER: list[str] = ["solopreneur", "startup", "sme", "enterprise"]

# Maps feature key to { label, description, min_tier }
FEATURE_ACCESS: dict[str, dict[str, str]] = {
    "workflows": {
        "label": "Workflow Engine",
        "description": "Automate multi-step business processes",
        "min_tier": "startup",
    },
    "sales": {
        "label": "Sales Pipeline & CRM",
        "description": "Track deals, manage pipeline, forecast revenue",
        "min_tier": "startup",
    },
    "reports": {
        "label": "Reports",
        "description": "Generate and view business reports",
        "min_tier": "startup",
    },
    "approvals": {
        "label": "Approvals",
        "description": "Multi-step approval workflows",
        "min_tier": "startup",
    },
    "compliance": {
        "label": "Compliance Suite",
        "description": "Audit trails, risk management, regulatory compliance",
        "min_tier": "sme",
    },
    "finance-forecasting": {
        "label": "Financial Forecasting",
        "description": "Revenue projections and financial modeling",
        "min_tier": "sme",
    },
    "custom-workflows": {
        "label": "Custom Workflows",
        "description": "Build custom workflow templates",
        "min_tier": "enterprise",
    },
    "governance": {
        "label": "SSO & Governance",
        "description": "Single sign-on and enterprise governance controls",
        "min_tier": "enterprise",
    },
}


def is_feature_allowed(feature_key: str, user_tier: str) -> bool:
    """Check if a user's tier meets the minimum requirement for a feature."""
    feature = FEATURE_ACCESS.get(feature_key)
    if not feature:
        return True  # Unknown features are ungated
    min_tier = feature["min_tier"]
    try:
        return TIER_ORDER.index(user_tier) >= TIER_ORDER.index(min_tier)
    except ValueError:
        return False  # Unknown tier = no access


def get_required_tier(feature_key: str) -> str | None:
    """Get the minimum tier required for a feature."""
    feature = FEATURE_ACCESS.get(feature_key)
    return feature["min_tier"] if feature else None
```

**Create `app/middleware/feature_gate.py`:**

A FastAPI dependency factory that checks persona tier before allowing endpoint execution.

```python
"""Feature gate middleware for persona tier enforcement.

Usage in routers:
    from app.middleware.feature_gate import require_feature

    @router.get("/audits")
    async def list_audits(
        request: Request,
        user_id: str = Depends(get_current_user_id),
        _gate: None = Depends(require_feature("compliance")),
    ):
"""

from __future__ import annotations

import logging
from typing import Any, Callable

from fastapi import Depends, HTTPException, Request
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer

from app.config.feature_gating import FEATURE_ACCESS, is_feature_allowed, get_required_tier
from app.personas.runtime import resolve_effective_persona
from app.routers.onboarding import get_current_user_id

logger = logging.getLogger(__name__)
security = HTTPBearer()


def require_feature(feature_key: str) -> Callable[..., Any]:
    """Create a FastAPI dependency that gates an endpoint by feature key.
    
    Returns 403 with upgrade message if the user's persona tier is below
    the feature's minimum tier requirement.
    """
    async def _check_feature_gate(
        request: Request,
        user_id: str = Depends(get_current_user_id),
    ) -> None:
        persona = await resolve_effective_persona(
            user_id=user_id,
            request=request,
        )
        
        if not persona:
            # No persona = treat as lowest tier (solopreneur)
            persona = "solopreneur"
        
        if not is_feature_allowed(feature_key, persona):
            required = get_required_tier(feature_key)
            feature = FEATURE_ACCESS.get(feature_key, {})
            label = feature.get("label", feature_key)
            raise HTTPException(
                status_code=403,
                detail={
                    "error": "feature_gated",
                    "message": f"{label} requires {required} tier or higher. Your current tier is {persona}.",
                    "feature": feature_key,
                    "current_tier": persona,
                    "required_tier": required,
                    "upgrade_url": "/dashboard/billing",
                },
            )
    
    return _check_feature_gate
```

The 403 response body includes structured JSON with `error`, `message`, `feature`, `current_tier`, `required_tier`, and `upgrade_url` so the frontend can display a rich upgrade prompt.

IMPORTANT: The dependency resolves persona from the request (cookie/header via resolve_effective_persona) and falls back to DB lookup if needed. This is the same pattern used by the rate limiter.

IMPORTANT: Do NOT import from the rate_limiter module — keep feature gating independent. Both use resolve_effective_persona but through separate dependency chains.

Apply the `require_feature` dependency to existing routers that serve gated features. The dependency must be added to EVERY endpoint in each gated router.

**Pattern for applying the gate:**

For each router, add `require_feature` as a dependency on the router itself (not individual endpoints) if the entire router is gated. If only some endpoints should be gated, add it to those specific endpoints.

**Preferred approach — router-level dependency:**

The cleanest way is to add the dependency at the router level so ALL endpoints in the router are gated:

```python
from app.middleware.feature_gate import require_feature

router = APIRouter(
    prefix="/workflows",
    tags=["Workflows"],
    dependencies=[Depends(require_feature("workflows"))],
)
```

This ensures every endpoint under `/workflows` is gated — no risk of forgetting an individual endpoint.
