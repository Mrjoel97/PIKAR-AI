**Create `frontend/src/components/dashboard/GatedPage.tsx`:**

A wrapper component that checks feature access and renders either the child page content or a full-page UpgradePrompt:

```typescript
'use client';

interface GatedPageProps {
  featureKey: FeatureKey;
  children: React.ReactNode;
}

export function GatedPage({ featureKey, children }: GatedPageProps) {
  const gate = useFeatureGate(featureKey);
  
  if (gate.isLoading) {
    // Render loading shimmer (same pattern as PersonaDashboardLayout)
    return <PremiumShell>...shimmer...</PremiumShell>;
  }
  
  if (!gate.allowed) {
    return (
      <PremiumShell>
        <div className="flex items-center justify-center min-h-[60vh]">
          <UpgradePrompt featureKey={featureKey} variant="page" />
        </div>
      </PremiumShell>
    );
  }
  
  return <>{children}</>;
}
```

Import `PremiumShell` from `@/components/layout/PremiumShell`, `useFeatureGate` from `@/hooks/useFeatureGate`, `UpgradePrompt` from `@/components/ui/UpgradePrompt`.
