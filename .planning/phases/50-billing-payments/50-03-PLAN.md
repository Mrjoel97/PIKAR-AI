<objective>
Replace the current placeholder churn metric (churn_pending / active_count) with a real time-windowed churn rate and compute MRR directly from the subscriptions table (not just the Stripe API pass-through), so the admin billing dashboard shows meaningful metrics even when Stripe is unreachable. BILL-04.

Purpose: The existing /admin/billing/summary endpoint at app/routers/admin/billing.py returns churn_rate = churn_pending/active_count, which is actually the "will-not-renew ratio" — not real churn. Real churn = (canceled in period) / (active at start of period). MRR is currently sourced from a Stripe integration proxy that can fail or return zero; the admin dashboard should also compute MRR from the authoritative subscriptions.price_id × tier price map so it is always available. After Plan 50-01 hardens the webhook, the subscriptions table becomes the single source of truth for subscription state, and this plan uses it.

Output: New BillingMetricsService with DB-native compute_mrr/compute_churn_rate, updated router, full test coverage.
</objective>

    ```python
    """BillingMetricsService — DB-native revenue and churn computation.

    Plan 50-03: Replaces the placeholder churn metric with a real time-windowed
    churn rate. Computes MRR from the subscriptions table (authoritative state
    written by the Stripe webhook handler hardened in Plan 50-01) so the admin
    dashboard shows meaningful numbers even when Stripe is unreachable.
    """
    from __future__ import annotations
    from datetime import datetime, timedelta, timezone
    from typing import Any, ClassVar
    from app.services.base_service import BaseService
    from app.services.supabase_async import execute_async

    class BillingMetricsService(BaseService):
        TIER_PRICES: ClassVar[dict[str, float]] = {
            "free": 0.0,
            "solopreneur": 99.0,
            "startup": 297.0,
            "sme": 597.0,
            "enterprise": 0.0,  # Custom pricing — excluded from MRR
        }

        async def compute_mrr(self) -> dict[str, float]:
            result = await execute_async(
                self.client.table("subscriptions").select("tier").eq("is_active", True),
                op_name="billing_metrics.compute_mrr",
            )
            rows = result.data or []
            mrr = sum(self.TIER_PRICES.get(row.get("tier", "free"), 0.0) for row in rows)
            return {"mrr": round(mrr, 2), "arr": round(mrr * 12, 2)}

        async def compute_churn_rate(self, days: int = 30) -> dict[str, Any]:
            window_start = datetime.now(tz=timezone.utc) - timedelta(days=days)
            iso = window_start.isoformat()

            canceled_result = await execute_async(
                self.client.table("subscriptions")
                    .select("user_id", count="exact")
                    .eq("is_active", False)
                    .gte("updated_at", iso),
                op_name="billing_metrics.compute_churn_rate.canceled",
            )
            canceled_in_period = canceled_result.count or 0

            active_result = await execute_async(
                self.client.table("subscriptions").select("user_id", count="exact").eq("is_active", True),
                op_name="billing_metrics.compute_churn_rate.active",
            )
            current_active = active_result.count or 0
            active_at_start = current_active + canceled_in_period

            churn_rate = (canceled_in_period / active_at_start) if active_at_start > 0 else 0.0
            return {
                "churn_rate": round(churn_rate, 4),
                "canceled_in_period": canceled_in_period,
                "active_at_start": active_at_start,
                "window_days": days,
    ```

    ```python
    # Add import
    from app.services.billing_metrics_service import BillingMetricsService

    @router.get("/billing/summary")
    @limiter.limit("120/minute")
    async def get_billing_summary(
        request: Request,
        admin_user: dict = Depends(require_admin),
        days: int = 30,
        include_trend: bool = False,
    ) -> dict[str, Any]:
        # 1. Plan distribution (existing — unchanged)
        client = get_service_client()
        sub_query = client.table("subscriptions").select("tier, is_active, will_renew, billing_issue_at")
        sub_result = await execute_async(sub_query, op_name="billing.summary.subscriptions")
        sub_rows = sub_result.data or []

        # 2. Plan distribution + legacy churn_pending (unchanged)
        active_tiers: Counter = Counter()
        churn_pending = 0
        billing_issues = 0
        for row in sub_rows:
            if row.get("is_active"):
                active_tiers[row.get("tier", "unknown")] += 1
                if not row.get("will_renew"):
                    churn_pending += 1
                if row.get("billing_issue_at") is not None:
                    billing_issues += 1
        total_active = sum(active_tiers.values())
        plan_distribution = [{"tier": t, "count": c} for t, c in sorted(active_tiers.items())]
        has_subscriptions = bool(sub_rows)

        # 3. NEW: DB-native MRR + real churn via BillingMetricsService
        metrics_svc = BillingMetricsService()
        mrr_data = await metrics_svc.compute_mrr()
        churn_data = await metrics_svc.compute_churn_rate(days=days)

        mrr = mrr_data["mrr"]
        arr = mrr_data["arr"]
        churn_rate = churn_data["churn_rate"]
        canceled_in_period = churn_data["canceled_in_period"]

        # 4. Stripe cross-check (non-fatal)
        stripe_configured = False
        cfg = await _get_integration_config("stripe")
        if not isinstance(cfg, dict):
            api_key, config, _ = cfg
            try:
                stripe_data = await IntegrationProxyService.call(
                    provider="stripe", operation="get_metrics",
                    api_key=api_key, config=config, params={},
                    fetch_fn=_fetch_stripe_metrics,
                )
                live_mrr = stripe_data.get("mrr", 0)
                stripe_configured = True
                if live_mrr and abs(live_mrr - mrr) / max(live_mrr, 1) > 0.10:
                    logger.warning(
                        "Billing MRR variance: db=%s stripe=%s", mrr, live_mrr
                    )
            except Exception as exc:
                logger.warning("Stripe metrics fetch failed: %s", exc)

        # 5. data_source
        if not has_subscriptions:
            data_source = "no_data"
        elif stripe_configured:
            data_source = "live"
        else:
            data_source = "db_only"

        # 6. Build response
        response: dict[str, Any] = {
            "mrr": mrr,
            "arr": arr,
            "churn_rate": churn_rate,
            "canceled_in_period": canceled_in_period,
            "active_subscriptions": total_active,
            "plan_distribution": plan_distribution,
            "churn_pending": churn_pending,
            "billing_issues": billing_issues,
            "data_source": data_source,
            "days": days,
        }
        if include_trend:
            response["churn_trend"] = await metrics_svc.compute_churn_trend(days=days)
        return response
    ```
