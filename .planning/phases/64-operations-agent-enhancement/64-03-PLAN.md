       ```sql
       -- Vendor/SaaS subscription tracking for Operations Agent
       CREATE TABLE IF NOT EXISTS vendor_subscriptions (
           id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
           user_id UUID NOT NULL,
           name TEXT NOT NULL,
           category TEXT NOT NULL DEFAULT 'other',
           -- Categories: project_management, communication, analytics, marketing,
           --   design, development, crm, accounting, storage, security, other
           monthly_cost NUMERIC(10,2) NOT NULL DEFAULT 0,
           billing_cycle TEXT NOT NULL DEFAULT 'monthly',
           -- Cycles: monthly, quarterly, annual
           annual_cost NUMERIC(10,2),
           renewal_date DATE,
           trial_end_date DATE,
           is_active BOOLEAN DEFAULT true,
           notes TEXT,
           integration_provider TEXT,
           -- Links to integration_credentials.provider if applicable
           created_at TIMESTAMPTZ DEFAULT now(),
           updated_at TIMESTAMPTZ DEFAULT now()
       );

       CREATE INDEX IF NOT EXISTS idx_vendor_subscriptions_user
           ON vendor_subscriptions(user_id);
       CREATE INDEX IF NOT EXISTS idx_vendor_subscriptions_trial
           ON vendor_subscriptions(trial_end_date)
           WHERE trial_end_date IS NOT NULL AND is_active = true;

       ALTER TABLE vendor_subscriptions ENABLE ROW LEVEL SECURITY;

       DO $$ BEGIN
           CREATE POLICY "Users manage their own subscriptions"
               ON vendor_subscriptions
               USING (user_id = (SELECT auth.uid()))
               WITH CHECK (user_id = (SELECT auth.uid()));
       EXCEPTION WHEN duplicate_object THEN NULL;
       END $$;

       DO $$ BEGIN
           CREATE POLICY "Service role manages all vendor subscriptions"
               ON vendor_subscriptions
               FOR ALL
               TO service_role
               USING (true)
               WITH CHECK (true);
       EXCEPTION WHEN duplicate_object THEN NULL;
       END $$;

       CREATE TRIGGER update_vendor_subscriptions_updated_at
           BEFORE UPDATE ON vendor_subscriptions
           FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
       ```

    2. Create `app/services/vendor_cost_service.py`:
       - Class `VendorCostService` extending `BaseService`.

       - Method `async add_subscription(user_id, name, category, monthly_cost, billing_cycle="monthly", annual_cost=None, renewal_date=None, trial_end_date=None, notes=None, integration_provider=None) -> dict`:
         - Insert into vendor_subscriptions.
         - If billing_cycle is "annual" and annual_cost is provided but monthly_cost is 0, compute monthly_cost = annual_cost / 12.
         - Return the inserted row.

       - Method `async list_subscriptions(user_id, active_only=True) -> dict`:
         - Query vendor_subscriptions for user.
         - Compute total_monthly_cost = sum of monthly_cost for active subscriptions.
         - Return `{subscriptions: [...], total_monthly_cost, total_annual_cost, count}`.

       - Method `async update_subscription(user_id, subscription_id, **updates) -> dict`:
         - Update the subscription row.
         - Return updated row.

       - Method `async delete_subscription(user_id, subscription_id) -> dict`:
         - Delete (or soft-delete by setting is_active=false) the subscription.
         - Return success status.

       - Method `async check_trial_expiries(user_id, days_ahead=7) -> list[dict]`:
         - Query vendor_subscriptions WHERE trial_end_date IS NOT NULL AND trial_end_date BETWEEN now() AND now() + interval 'N days' AND is_active = true.
         - Return list with days_remaining computed.

       - Method `async get_cost_summary(user_id) -> dict`:
         - Call list_subscriptions to get all active.
         - Group by category, compute per-category totals.
         - Detect consolidation opportunities: categories with 2+ subscriptions.
         - Generate suggestions: "You have {N} {category} tools ({names}). Consider consolidating to reduce costs by ~${potential_savings}/month."
         - Check for trial expiries (call check_trial_expiries).
         - Return `{total_monthly, total_annual_estimate, by_category: {...}, trial_expiring: [...], consolidation_suggestions: [...]}`.

       - Module-level convenience functions: `async def get_vendor_costs(user_id)`, `async def check_trial_expiries(user_id)`.
