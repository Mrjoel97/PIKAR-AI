```python
def _make_promoted_tool(tool_name: str, category: str, delegate_fn, description_template: str):
    """Create a promoted tool that delegates to a real function and tracks the event."""
    async def promoted_tool(**kwargs) -> dict:
        desc = description_template.format(**kwargs)
        result = await delegate_fn(**{k: v for k, v in kwargs.items() if k in delegate_fn.__code__.co_varnames})
        await track_event(
            event_name=tool_name, category=category,
            properties=json.dumps({k: str(v) for k, v in kwargs.items()}, default=str),
        )
        return {"success": True, "result": result, "tool": tool_name}
    promoted_tool.__name__ = f"promoted_{tool_name}"
    promoted_tool.__qualname__ = f"promoted_{tool_name}"
    return promoted_tool
```
