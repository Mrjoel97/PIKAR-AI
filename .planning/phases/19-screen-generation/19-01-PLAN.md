    def _build_generation_prompt(screen_name: str, page_slug: str, design_system: dict) -> str:
        parts = [f"Generate a {screen_name} page"]
        if design_system:
            colors = design_system.get("colors", [])
            if colors:
                palette = ", ".join(c.get("hex", "") for c in colors if isinstance(c, dict))
                parts.append(f"Color palette: {palette}")

        return ". ".join(parts)
