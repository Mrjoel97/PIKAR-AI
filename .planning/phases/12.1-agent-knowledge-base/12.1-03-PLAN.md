```
GET  /admin/knowledge/stats
  Response: { total_entries: number, total_embeddings: number, by_agent: Record<string, number>, storage_bytes: number }

GET  /admin/knowledge/entries?agent_scope=X&status=Y&limit=50&offset=0
  Response: { data: KnowledgeEntry[], count: number }
  KnowledgeEntry: { id, filename, file_type, mime_type, agent_scope, uploaded_by, status, chunk_count, file_size_bytes, created_at }

GET  /admin/knowledge/entries/{entry_id}
  Response: KnowledgeEntry

POST /admin/knowledge/upload  (multipart/form-data)
  Body: file (File), agent_scope (string, optional), uploaded_by (string)
  Response: { entry_id, chunk_count?, status, description?, message? }

DELETE /admin/knowledge/entries/{entry_id}
  Response: { deleted: true }
```
