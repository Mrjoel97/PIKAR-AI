# Collect IDs of steps that need resetting
ids_to_reset = [
    step["id"]
    for step in steps[last_completed_idx + 1:]
    if step.get("status") in ("failed", "skipped", "cancelled")
]
reset_count = len(ids_to_reset)

if ids_to_reset:
    await client.table("workflow_steps").update(
        {
            "status": "pending",
            "error_message": None,
            "started_at": None,
            "completed_at": None,

**2. `app/persistence/supabase_session_service.py` -- `rollback_session` method (~line 905-912):**

Replace the per-event UPDATE loop with a single `.in_()` batch:

```python
# Mark superseded events (batch)
if rollback_event_id and events_to_supersede.data:
    supersede_ids = [evt["id"] for evt in events_to_supersede.data]
    await self._execute_with_retry(
        client.table(self.events_table)
        .update({"superseded_by": rollback_event_id})

    )
```

**3. `app/persistence/supabase_session_service.py` -- `fork_session` method (~line 809-811):**

Replace the per-event `append_event` loop with a single bulk insert. Because `append_event` uses the `insert_session_event` RPC for atomic versioning, forking needs a direct bulk insert instead (forked events get new sequential versions in the new session):

```python
# Copy events to new session (bulk insert)
if source.events:
    client = await self._get_client()
    user_id_str = self._ensure_uuid_str(user_id)
    bulk_rows = []
    for idx, event in enumerate(source.events):
        event_data = event.model_dump(mode="json")
        bulk_rows.append({
            "app_name": app_name,
            "user_id": user_id_str,
            "session_id": new_session_id,
            "event_data": event_data,
            "event_index": idx,
            "version": idx + 1,
            "operation": "fork",
        })
    await self._execute_with_retry(
        client.table(self.events_table).insert(bulk_rows)
    )
    # Update session version to match final event
    await self._execute_with_retry(
        client.table(self.sessions_table)
        .update({"current_version": len(source.events), "updated_at": "now()"})
        .eq("app_name", app_name)
        .eq("user_id", user_id_str)
        .eq("session_id", new_session_id)
    )
```

Also update the new_session object's events list in-memory: `new_session.events = list(source.events)`.
