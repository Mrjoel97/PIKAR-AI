GET /admin/analytics/summary?days=30 response:
```typescript
interface AnalyticsSummaryResponse {
  usage_trends: Array<{
    stat_date: string;
    dau: number;
    mau: number;
    messages: number;
    workflows: number;
  }>;
  agent_effectiveness: Array<{
    agent_name: string;
    success_rate: number;    // 0-100 percentage
    avg_duration_ms: number;
    total_calls: number;
  }>;
  feature_usage: {
    by_tool: Array<{ tool_name: string; call_count: number }>;
    by_category: Array<{ category: string; event_count: number }>;
  };
  config_status: {
    permission_counts: { auto: number; confirm: number; blocked: number };
    last_config_change: string | null;  // ISO timestamp
  };
  days: number;
  data_source: 'aggregated' | 'no_data';
}
```

```typescript
const REFRESH_INTERVAL_MS = 30_000;  // Use 60_000 for analytics
const supabase = createClient();
const API_URL = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000';
const fetchStatus = useCallback(async () => {
  const { data: { session } } = await supabase.auth.getSession();
  // ... fetch with Bearer token
}, [supabase, API_URL]);
useEffect(() => {
  fetchStatus();
  const interval = setInterval(fetchStatus, REFRESH_INTERVAL_MS);
  return () => clearInterval(interval);
}, [fetchStatus]);
```
